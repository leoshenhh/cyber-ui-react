# Install

`npm install cyber-ui-react --save`

# Usage
```tsx

import React from 'react';
import ReactDOM from 'react-dom';
import {Cyber} from 'cyber-ui-react';

ReactDOM.render((
  <div>
    <Cyber>CyberUI</Cyber>
  </div>
), document.getElementById('root'));

```
# Browser Support
Modern browsers and Internet Explorer 10+.
